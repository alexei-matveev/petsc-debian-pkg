  2: static char help[] = "Tests MatTranspose(), MatNorm(), MatAXPY() and MatAYPX().\n\n";

  4: #include <petscmat.h>

  8: int main(int argc,char **argv)
  9: {
 10:   Mat              mat,tmat = 0;
 11:   PetscInt         m = 7,n,i,j,rstart,rend,rect = 0;
 12:   PetscErrorCode   ierr;
 13:   PetscMPIInt      size,rank;
 14:   PetscBool        flg;
 15:   PetscScalar      v, alpha;
 16:   PetscReal        normf,normi,norm1;

 18:   PetscInitialize(&argc,&argv,(char*)0,help);
 19:   PetscViewerSetFormat(PETSC_VIEWER_STDOUT_WORLD,PETSC_VIEWER_ASCII_COMMON);
 20:   PetscOptionsGetInt(PETSC_NULL,"-m",&m,PETSC_NULL);
 21:   MPI_Comm_rank(PETSC_COMM_WORLD,&rank);
 22:   MPI_Comm_size(PETSC_COMM_WORLD,&size);
 23:   n = m;
 24:   PetscOptionsHasName(PETSC_NULL,"-rectA",&flg);
 25:   if (flg) {n += 2; rect = 1;}
 26:   PetscOptionsHasName(PETSC_NULL,"-rectB",&flg);
 27:   if (flg) {n -= 2; rect = 1;}

 29:   /* ------- Assemble matrix, test MatValid() --------- */
 30:   MatCreate(PETSC_COMM_WORLD,&mat);
 31:   MatSetSizes(mat,PETSC_DECIDE,PETSC_DECIDE,m,n);
 32:   MatSetFromOptions(mat);
 33:   MatGetOwnershipRange(mat,&rstart,&rend);
 34:   for (i=rstart; i<rend; i++) {
 35:     for (j=0; j<n; j++) {
 36:       v=10.0*i+j;
 37:       MatSetValues(mat,1,&i,1,&j,&v,INSERT_VALUES);
 38:     }
 39:   }
 40:   MatAssemblyBegin(mat,MAT_FINAL_ASSEMBLY);
 41:   MatAssemblyEnd(mat,MAT_FINAL_ASSEMBLY);

 43:   /* ----------------- Test MatNorm()  ----------------- */
 44:   MatNorm(mat,NORM_FROBENIUS,&normf);
 45:   MatNorm(mat,NORM_1,&norm1);
 46:   MatNorm(mat,NORM_INFINITY,&normi);
 47:   PetscPrintf(PETSC_COMM_WORLD,"original A: Frobenious norm = %G, one norm = %G, infinity norm = %G\n",
 48:                      normf,norm1,normi);
 49:   MatView(mat,PETSC_VIEWER_STDOUT_WORLD);

 51:   /* --------------- Test MatTranspose()  -------------- */
 52:   PetscOptionsHasName(PETSC_NULL,"-in_place",&flg);
 53:   if (!rect && flg) {
 54:     MatTranspose(mat,MAT_REUSE_MATRIX,&mat);   /* in-place transpose */
 55:     tmat = mat; mat = 0;
 56:   } else {      /* out-of-place transpose */
 57:     MatTranspose(mat,MAT_INITIAL_MATRIX,&tmat);
 58:   }

 60:   /* ----------------- Test MatNorm()  ----------------- */
 61:   /* Print info about transpose matrix */
 62:   MatNorm(tmat,NORM_FROBENIUS,&normf);
 63:   MatNorm(tmat,NORM_1,&norm1);
 64:   MatNorm(tmat,NORM_INFINITY,&normi);
 65:   PetscPrintf(PETSC_COMM_WORLD,"B = A^T: Frobenious norm = %G, one norm = %G, infinity norm = %G\n",
 66:                      normf,norm1,normi);
 67:   MatView(tmat,PETSC_VIEWER_STDOUT_WORLD);

 69:   /* ----------------- Test MatAXPY(), MatAYPX()  ----------------- */
 70:   if (mat && !rect) {
 71:     alpha = 1.0;
 72:     PetscOptionsGetScalar(PETSC_NULL,"-alpha",&alpha,PETSC_NULL);
 73:     PetscPrintf(PETSC_COMM_WORLD,"MatAXPY:  B = B + alpha * A\n");
 74:     MatAXPY(tmat,alpha,mat,DIFFERENT_NONZERO_PATTERN);
 75:     MatView(tmat,PETSC_VIEWER_STDOUT_WORLD);

 77:     PetscPrintf(PETSC_COMM_WORLD,"MatAYPX:  B = alpha*B + A\n");
 78:     MatAYPX(tmat,alpha,mat,DIFFERENT_NONZERO_PATTERN);
 79:     MatView(tmat,PETSC_VIEWER_STDOUT_WORLD);
 80:   }

 82:   {
 83:     Mat C;
 84:     alpha = 1.0;
 85:     PetscPrintf(PETSC_COMM_WORLD,"MatAXPY:  C = C + alpha * A, C=A, SAME_NONZERO_PATTERN\n");
 86:     MatDuplicate(mat,MAT_COPY_VALUES,&C);
 87:     MatAXPY(C,alpha,mat,SAME_NONZERO_PATTERN);
 88:     MatView(C,PETSC_VIEWER_STDOUT_WORLD);
 89:     MatDestroy(&C);
 90:   }

 92:   {
 93:     Mat matB;
 94:     /* get matB that has nonzeros of mat in all even numbers of row and col */
 95:     MatCreate(PETSC_COMM_WORLD,&matB);
 96:     MatSetSizes(matB,PETSC_DECIDE,PETSC_DECIDE,m,n);
 97:     MatSetFromOptions(matB);
 98:     MatGetOwnershipRange(matB,&rstart,&rend);
 99:     if (rstart % 2 != 0) rstart++;
100:     for (i=rstart; i<rend; i += 2) {
101:       for (j=0; j<n; j += 2) {
102:         v=10.0*i+j;
103:         MatSetValues(matB,1,&i,1,&j,&v,INSERT_VALUES);
104:       }
105:     }
106:     MatAssemblyBegin(matB,MAT_FINAL_ASSEMBLY);
107:     MatAssemblyEnd(matB,MAT_FINAL_ASSEMBLY);
108:     PetscPrintf(PETSC_COMM_WORLD," A: original matrix:\n");
109:     MatView(mat,PETSC_VIEWER_STDOUT_WORLD);
110:     PetscPrintf(PETSC_COMM_WORLD," B(a subset of A):\n");
111:     MatView(matB,PETSC_VIEWER_STDOUT_WORLD);
112:     PetscPrintf(PETSC_COMM_WORLD,"MatAXPY:  B = B + alpha * A, SUBSET_NONZERO_PATTERN\n");
113:     MatAXPY(mat,alpha,matB,SUBSET_NONZERO_PATTERN);
114:     MatView(mat,PETSC_VIEWER_STDOUT_WORLD);
115:     MatDestroy(&matB);
116:   }

118:   /* Free data structures */
119:   if (mat)  {MatDestroy(&mat);}
120:   if (tmat) {MatDestroy(&tmat);}

122:   PetscFinalize();
123:   return 0;
124: }
